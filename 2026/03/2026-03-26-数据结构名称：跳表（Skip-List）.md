```python
# 数据结构名称：跳表（Skip List）
# 原理说明：
# 跳表是一种可以用于有序数据集合快速查找的数据结构，它通过在链表中建立多级索引，
# 使得在查找、插入、删除等操作的时间复杂度接近O(log n)。
# 每个节点包含一个值和多个指针，指针指向不同层级的下一个节点。层级越高，跳过的节点越多，
# 从而实现快速定位。跳表的层级结构是随机生成的，通常用概率来决定节点是否上浮到更高级别。
# 跳表的核心思想是牺牲空间复杂度来换取时间复杂度的优化，一般空间复杂度是O(n log n)。

import random

class SkipList:
    class Node:
        def __init__(self, value, level=0):
            self.value = value
            self.forward = [None] * (level + 1)  # 每个节点包含多个指针

    def __init__(self, max_level=16, p=0.5):
        self.max_level = max_level  # 最大层级
        self.p = p  # 概率，决定节点是否升级
        self.head = self.Node(None, self.max_level)  # 跳表头节点
        self.level = 0  # 当前跳表最大层级

    def _random_level(self):
        # 随机生成节点的层级，最大不超过max_level
        level = 1
        while random.random() < self.p and level < self.max_level:
            level += 1
        return level

    def insert(self, value):
        # 插入元素
        update = [None] * self.max_level
        node = self.head
        for i in range(self.level, -1, -1):
            while node.forward[i] and node.forward[i].value < value:
                node = node.forward[i]
            update[i] = node
        node = self.Node(value, self._random_level())
        for i in range(node.level):
            node.forward[i] = update[i].forward[i]
            update[i].forward[i] = node
        if node.level > self.level:
            self.level = node.level

    def delete(self, value):
        # 删除元素
        update = [None] * self.max_level
        node = self.head
        for i in range(self.level, -1, -1):
            while node.forward[i] and node.forward[i].value < value:
                node = node.forward[i]
            update[i] = node
        if node.forward[0] and node.forward[0].value == value:
            for i in range(self.level):
                if update[i].forward[i] != node:
                    break
                update[i].forward[i] = node.forward[i]
            while self.level > 0 and self.head.forward[self.level - 1] is None:
                self.level -= 1

    def find(self, value):
        # 查找元素
        node = self.head
        for i in range(self.level - 1, -1, -1):
            while node.forward[i] and node.forward[i].value < value:
                node = node.forward[i]
        if node.forward[0] and node.forward[0].value == value:
            return node.forward[0]
        return None

    def display_list(self):
        # 展示跳表的结构（调试用）
        for i in range(self.level):
            print(f"Level {i}:", end='')
            node = self.head.forward[i]
            while node:
                print(f" -> {node.value}", end='')
                node = node.forward[i]
            print()

# 使用示例和测试代码
if __name__ == "__main__":
    skip_list = SkipList()
    values = [3, 6, 7, 9, 12, 15, 18, 20]

    print("Inserting values:", values)
    for v in values:
        skip_list.insert(v)
    skip_list.display_list()

    print("\nFinding 12:", skip_list.find(12).value if skip_list.find(12) else "Not found")
    print("Finding 10:", skip_list.find(10) if skip_list.find(10) else "Not found")

    print("\nDeleting 12:")
    skip_list.delete(12)
    skip_list.display_list()

    print("\nFinding 12 after deletion:", skip_list.find(12) if skip_list.find(12) else "Not found")
```

### **讲解**

- **跳表（Skip List）** 是一种基于概率的数据结构，适用于需要频繁查找、插入和删除的有序集合。
- **节点结构**：每个节点保存一个值，并且包含多个指针，用于在不同层级上跳转。层级越高，跳过的节点越多，查找效率越高。
- **随机层级**：为保证跳表的效率和平衡性，每个节点的层级由随机函数决定。默认情况下，每个节点有50%的概率升级到下一层，最大不超过给定的 `max_level`。
- **插入过程**：需要找到插入位置的前驱节点，并在所有层级上更新指针，使新节点成为该层级的下一个节点。
- **删除过程**：找到要删除节点的前驱节点，并断开所有层级的指针链接。若某层级的头节点没有后续元素，则层级递减。
- **查找过程**：从最高层级开始，尽可能跳过更多的节点，最终在最低层级找到目标值。

跳表是实现有序集合的一种高效方式，常用于数据库系统中以提高查找效率，如 Redis 的有序集合（ZSET）就是基于跳表实现的。
