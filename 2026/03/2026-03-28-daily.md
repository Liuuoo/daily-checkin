"""
红黑树 (Red-Black Tree)
原理说明：
红黑树是一种自平衡的二叉搜索树，它通过以下规则保持平衡：
1. 每个节点是红色或黑色
2. 根节点是黑色
3. 所有叶子节点（NIL节点）是黑色
4. 红色节点的两个子节点都是黑色（不能有连续的红色节点）
5. 从任一节点到其每个叶子节点的所有路径包含相同数量的黑色节点

这些约束确保了最坏情况下的操作时间复杂度为O(log n)
"""

class Node:
    """红黑树节点类"""
    def __init__(self, key, value=None, color='RED'):
        self.key = key
        self.value = value
        self.left = None
        self.right = None
        self.parent = None
        self.color = color  # 'RED' 或 'BLACK'


class RedBlackTree:
    """红黑树实现"""
    
    def __init__(self):
        self.NIL = Node(None, None, 'BLACK')  # 哨兵节点
        self.root = self.NIL
    
    def left_rotate(self, x):
        """左旋操作"""
        y = x.right
        x.right = y.left
        
        if y.left != self.NIL:
            y.left.parent = x
        
        y.parent = x.parent
        
        if x.parent == self.NIL:
            self.root = y
        elif x == x.parent.left:
            x.parent.left = y
        else:
            x.parent.right = y
        
        y.left = x
        x.parent = y
    
    def right_rotate(self, y):
        """右旋操作"""
        x = y.left
        y.left = x.right
        
        if x.right != self.NIL:
            x.right.parent = y
        
        x.parent = y.parent
        
        if y.parent == self.NIL:
            self.root = x
        elif y == y.parent.right:
            y.parent.right = x
        else:
            y.parent.left = x
        
        x.right = y
        y.parent = x
    
    def insert_fixup(self, z):
        """插入后修复红黑树性质"""
        while z.parent.color == 'RED':
            if z.parent == z.parent.parent.left:
                y = z.parent.parent.right  # 叔节点
                
                if y.color == 'RED':
                    # Case 1: 叔节点为红色
                    z.parent.color = 'BLACK'
                    y.color = 'BLACK'
                    z.parent.parent.color = 'RED'
                    z = z.parent.parent
                else:
                    if z == z.parent.right:
                        # Case 2: z是右孩子
                        z = z.parent
                        self.left_rotate(z)
                    
                    # Case 3: z是左孩子
                    z.parent.color = 'BLACK'
                    z.parent.parent.color = 'RED'
                    self.right_rotate(z.parent.parent)
            else:
                # 对称情况
                y = z.parent.parent.left
                
                if y.color == 'RED':
                    z.parent.color = 'BLACK'
                    y.color = 'BLACK'
                    z.parent.parent.color = 'RED'
                    z = z.parent.parent
                else:
                    if z == z.parent.left:
                        z = z.parent
                        self.right_rotate(z)
                    
                    z.parent.color = 'BLACK'
                    z.parent.parent.color = 'RED'
                    self.left_rotate(z.parent.parent)
        
        self.root.color = 'BLACK'
    
    def insert(self, key, value=None):
        """插入键值对"""
        z = Node(key, value)
        z.left = self.NIL
        z.right = self.NIL
        
        y = self.NIL
        x = self.root
        
        # 找到插入位置
        while x != self.NIL:
            y = x
            if z.key < x.key:
                x = x.left
            else:
                x = x.right
        
        z.parent = y
        
        if y == self.NIL:
            self.root = z
        elif z.key < y.key:
            y.left = z
        else:
            y.right = z
        
        self.insert_fixup(z)
        return z
    
    def transplant(self, u, v):
        """用v子树替换u子树"""
        if u.parent == self.NIL:
            self.root = v
        elif u == u.parent.left:
            u.parent.left = v
        else:
            u.parent.right = v
        v.parent = u.parent
    
    def minimum(self, node):
        """找到子树的最小节点"""
        while node.left != self.NIL:
            node = node.left
        return node
    
    def delete_fixup(self, x):
        """删除后修复红黑树性质"""
        while x != self.root and x.color == 'BLACK':
            if x == x.parent.left:
                w = x.parent.right  # 兄弟节点
                
                if w.color == 'RED':
                    # Case 1: 兄弟节点为红色
                    w.color = 'BLACK'
                    x.parent.color = 'RED'
                    self.left_rotate(x.parent)
                    w = x.parent.right
                
                if w.left.color == 'BLACK' and w.right.color == 'BLACK':
                    # Case 2: 兄弟节点的两个子节点都是黑色
                    w.color = 'RED'
                    x = x.parent
                else:
                    if w.right.color == 'BLACK':
                        # Case 3: 兄弟节点的右子节点为黑色
                        w.left.color = 'BLACK'
                        w.color = 'RED'
                        self.right_rotate(w)
                        w = x.parent.right
                    
                    # Case 4: 兄弟节点的右子节点为红色
                    w.color = x.parent.color
                    x.parent.color = 'BLACK'
                    w.right.color = 'BLACK'
                    self.left_rotate(x.parent)
                    x = self.root
            else:
                # 对称情况
                w = x.parent.left
                
                if w.color == 'RED':
                    w.color = 'BLACK'
                    x.parent.color = 'RED'
                    self.right_rotate(x.parent)
                    w = x.parent.left
                
                if w.right.color == 'BLACK' and w.left.color == 'BLACK':
                    w.color = 'RED'
                    x = x.parent
                else:
                    if w.left.color == 'BLACK':
                        w.right.color = 'BLACK'
                        w.color = 'RED'
                        self.left_rotate(w)
                        w = x.parent.left
                    
                    w.color = x.parent.color
                    x.parent.color = 'BLACK'
                    w.left.color = 'BLACK'
                    self.right_rotate(x.parent)
                    x = self.root
        
        x.color = 'BLACK'
    
    def delete(self, key):
        """删除指定键的节点"""
        z = self.search(self.root, key)
        if z == self.NIL:
            return False
        
        y = z
        y_original_color = y.color
        
        if z.left == self.NIL:
            x = z.right
            self.transplant(z, z.right)
        elif z.right == self.NIL:
            x = z.left
            self.transplant(z, z.left)
        else:
            y = self.minimum(z.right)
            y_original_color = y.color
            x = y.right
            
            if y.parent == z:
                x.parent = y
            else:
                self.transplant(y, y.right)
                y.right = z.right
                y.right.parent = y
            
            self.transplant(z, y)
            y.left = z.left
            y.left.parent = y
            y.color = z.color
        
        if y_original_color == 'BLACK':
            self.delete_fixup(x)
        
        return True
    
    def search(self, node, key):
        """搜索指定键的节点"""
        while node != self.NIL and key != node.key:
            if key < node.key:
                node = node.left
            else:
                node = node.right
        return node
    
    def inorder_traversal(self, node, result=None):
        """中序遍历"""
        if result is None:
            result = []
        
        if node != self.NIL:
            self.inorder_traversal(node.left, result)
            result.append((node.key, node.value, node.color))
            self.inorder_traversal(node.right, result)
        
        return result
    
    def height(self, node):
        """计算树的高度"""
        if node == self.NIL:
            return 0
        return 1 + max(self.height(node.left), self.height(node.right))


# 使用示例和测试代码
if __name__ == "__main__":
    print("=== 红黑树测试 ===")
    
    # 创建红黑树
    rbt = RedBlackTree()
    
    # 插入测试
    test_data = [(10, 'A'), (20, 'B'), (30, 'C'), (15, 'D'), (25, 'E'), (5, 'F')]
    
    print("插入数据:", test_data)
    for key, value in test_data:
        rbt.insert(key, value)
    
    # 中序遍历
    inorder_result = rbt.inorder_traversal(rbt.root)
    print("\n中序遍历结果（键, 值, 颜色）:")
    for item in inorder_result:
        print(f"  {item}")
    
    # 搜索测试
    search_keys = [15, 30, 100]
    print("\n搜索测试:")
    for key in search_keys:
        node = rbt.search(rbt.root, key)
        if node != rbt.NIL:
            print(f"  键 {key}: 找到，值 = {node.value}")
        else:
            print(f"  键 {key}: 未找到")
    
    # 删除测试
    print("\n删除键 20:")
    if rbt.delete(20):
        print("  删除成功")
        inorder_after_delete = rbt.inorder_traversal(rbt.root)
        print("  删除后中序遍历:", [item[0] for item in inorder_after_delete])
    
    # 树高度
    print(f"\n树的高度: {rbt.height(rbt.root)}")
    
    # 性能测试
    print("\n=== 性能测试 ===")
    import random
    import time
    
    large_rbt = RedBlackTree()
    test_size = 10000
    test_keys = random.sample(range(test_size * 10), test_size)
    
    # 插入性能
    start = time.time()
    for key in test_keys:
        large_rbt.insert(key)
    insert_time = time.time() - start
    print(f"插入 {test_size} 个随机键耗时: {insert_time:.4f} 秒")
    
    # 搜索性能
    search_keys = random.sample(test_keys, 1000)
    start = time.time()
    for key in search_keys:
        large_rbt.search(large_rbt.root, key)
    search_time = time.time() - start

"""
复杂度分析：

时间复杂度：
1. 搜索 (search): O(log n)
   - 红黑树的高度最多为 2log(n+1)，因此搜索复杂度为 O(log n)
   
2. 插入 (insert): O(log n)
   - 普通BST插入: O(log n)
   - 修复操作: O(log n)（最多进行常数次旋转）
   
3. 删除 (delete): O(log n)
   - 普通BST删除: O(log n)
   - 修复操作: O(log n)（最多进行常数次旋转）

4. 遍历 (inorder_traversal): O(n)
   - 需要访问所有节点

空间复杂度：
1. 存储: O(n)
   - 每个节点需要存储键、值、颜色和三个指针
   
2. 递归栈: O(log n)
   - 遍历和高度计算的最大递归深度为树的高度

优势：
1. 保证最坏情况下的 O(log n) 操作复杂度
2. 相比AVL树，旋转操作更少，插入删除性能更好
3. 适合需要频繁插入删除的场景

劣势：
1. 实现复杂，容易出错
2. 相比普通BST有额外的存储开销（颜色位）
3. 常数因子比AVL树稍大

应用场景：
1. 数据库索引
2. 内存中的有序数据结构
3. Linux内核的进程调度
4. Java的TreeMap和TreeSet实现
"""
