```python
# 数据结构名称：Trie（前缀树）
# 原理说明：Trie 是一种树形数据结构，用于高效存储和查找字符串集合中的前缀。它的每个节点代表一个字符，从根节点到某个节点的路径代表一个字符串的前缀。Trie 的主要优势在于字符串的前缀查找效率高，尤其适用于字典、自动补全、拼写检查等场景。

class TrieNode:
    def __init__(self):
        # 子节点字典，键为字符，值为对应的子节点
        self.children = {}
        # 标记该节点是否为某个单词的结尾
        self.is_end = False

class Trie:
    def __init__(self):
        # 初始化根节点
        self.root = TrieNode()

    def insert(self, word: str) -> None:
        # 插入一个单词到 Trie 中
        node = self.root
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.is_end = True

    def search(self, word: str) -> bool:
        # 查找单词是否存在于 Trie 中
        node = self.root
        for char in word:
            if char not in node.children:
                return False
            node = node.children[char]
        return node.is_end

    def starts_with(self, prefix: str) -> bool:
        # 查找是否有单词以给定前缀开头
        node = self.root
        for char in prefix:
            if char not in node.children:
                return False
            node = node.children[char]
        return True

    def delete(self, word: str) -> None:
        # 删除一个单词，注意需要处理路径上的节点是否被其他单词共用
        def _delete(node, word, index):
            if index == len(word):
                if not node.is_end:
                    return False  # 单词不存在
                node.is_end = False  # 如果该节点是其他单词的前缀，则保留该节点
                return len(node.children) == 0  # 如果该节点无子节点，可以被删除
            char = word[index]
            child_node = node.children.get(char)
            if not child_node:
                return False  # 子节点不存在，无法删除
            # 递归删除子节点
            delete_child = _delete(child_node, word, index + 1)
            if delete_child:
                del node.children[char]
                return len(node.children) == 0
            return False

        _delete(self.root, word, 0)

    def get_words(self) -> list:
        # 获取 Trie 中所有插入的单词
        words = []

        def _get_words(node, prefix, words):
            if node.is_end:
                words.append(prefix)
            for char, child in node.children.items():
                _get_words(child, prefix + char, words)

        _get_words(self.root, "", words)
        return words

# 使用示例
if __name__ == "__main__":
    trie = Trie()
    # 插入单词
    trie.insert("apple")
    trie.insert("app")
    trie.insert("banana")
    trie.insert("bat")
    trie.insert("applet")

    # 查找单词
    print("Search 'apple':", trie.search("apple"))  # True
    print("Search 'app':", trie.search("app"))      # True
    print("Search 'apples':", trie.search("apples"))# False

    # 查找前缀
    print("Starts with 'app':", trie.starts_with("app"))  # True
    print("Starts with 'ban':", trie.starts_with("ban"))  # True
    print("Starts with 'apples':", trie.starts_with("apples"))  # False

    # 删除单词
    trie.delete("app")
    print("Search 'app' after deletion:", trie.search("app"))  # False
    print("Starts with 'app' after deletion:", trie.starts_with("app"))  # True (因为 'apple' 仍存在)

    # 获取所有单词
    print("All words in Trie:", trie.get_words())  # ['apple', 'banana', 'bat', 'applet']
```

### 复杂度分析

1. **插入操作 (insert)**  
   - **时间复杂度**: O(L), 其中 L 是单词的长度。在最坏情况下，每个字符都需要创建新的节点。
   - **空间复杂度**: O(L), 用于存储新插入的字符节点。

2. **搜索操作 (search)**  
   - **时间复杂度**: O(L), 需要遍历字典中每个字符对应的路径。
   - **空间复杂度**: O(1), 无额外空间使用。

3. **前缀查找 (starts_with)**  
   - **时间复杂度**: O(L), 遍历到给定前缀的末尾。
   - **空间复杂度**: O(1), 无额外空间使用。

4. **删除操作 (delete)**  
   - **时间复杂度**: O(L), 同样需要遍历到单词的末尾，同时可能需要递归地检查子节点是否可删除。
   - **空间复杂度**: O(1), 删除操作不额外分配空间，但可能回收部分节点。

5. **获取所有单词 (get_words)**  
   - **时间复杂度**: O(N), 其中 N 是 Trie 中所有字符的总数（即存储空间）。因为需要遍历所有可能的路径。
   - **空间复杂度**: O(N), 存储所有单词的输出列表。

### 适用场景

- **自动补全**（如搜索引擎、IDE 的智能提示）。
- **拼写检查**（检查输入是否存在于字典中）。
- **IP 路由查找**（长前缀匹配）。
- **单词频率统计**（如 Trie 可扩展为存储计数）。

### 优化建议

- **压缩 Trie**（Radix Tree）: 减少节点数量，适用于存储大量具有共同前缀的字符串。
- **使用数组代替字典**（若字符集固定，如仅小写字母）: 可进一步优化查找和插入效率。
- **添加计数功能**（支持单词出现次数）: 可以在每个节点增加一个 `count` 字段，用于统计某个前缀的出现次数或某个单词的出现频率。

Trie 在处理字符串集合的前缀查找和插入操作中非常高效，尤其在字符集较小或字符串长度较短时表现优异。
