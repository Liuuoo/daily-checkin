```python
# 实现的数据结构：Trie树（前缀树）
# Trie（发音为 "try"）是一种用于处理字符串的树状数据结构，用于高效存储和检索具有公共前缀的单词集合。
# 基本原理：每个节点代表一个字符，从根节点开始，每个子节点代表一个字符，通过路径查找单词。
# Trie非常适合实现自动补全、拼写检查、IP路由等应用场景。
# 每个节点包含一个字典（children），用于指向子节点，以及一个标志（is_end_of_word）表示该节点是否为某个单词的结尾。

class TrieNode:
    def __init__(self):
        # 存储子节点的字典，键为字符，值为对应的TrieNode
        self.children = {}
        # 标志该节点是否为某个单词的结尾
        self.is_end_of_word = False

class Trie:
    def __init__(self):
        # 初始化根节点
        self.root = TrieNode()

    def insert(self, word: str) -> None:
        # 插入一个单词到Trie中
        node = self.root
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.is_end_of_word = True

    def search(self, word: str) -> bool:
        # 在Trie中搜索一个单词是否存在
        node = self.root
        for char in word:
            if char not in node.children:
                return False
            node = node.children[char]
        return node.is_end_of_word

    def starts_with(self, prefix: str) -> bool:
        # 检查是否有单词以给定的前缀开头
        node = self.root
        for char in prefix:
            if char not in node.children:
                return False
            node = node.children[char]
        return True

    def delete(self, word: str) -> None:
        # 删除一个单词，注意要处理可能存在的多个单词共享路径的情况
        def _delete(node, word, index):
            if index == len(word):
                # 如果该节点不是任何单词的结尾，则不删除
                if not node.is_end_of_word:
                    return False
                node.is_end_of_word = False
                return len(node.children) == 0

            char = word[index]
            child_node = node.children.get(char)
            if not child_node:
                return False

            # 递归删除子节点
            if _delete(child_node, word, index + 1):
                # 如果子节点被删除且没有其他子节点，则删除当前节点
                del node.children[char]
                return len(node.children) == 0

            return False

        _delete(self.root, word, 0)

    def get_all_words(self) -> list:
        # 获取当前Trie中存储的所有单词
        words = []

        def _collect_words(node, current_word):
            if node.is_end_of_word:
                words.append(current_word)
            for char, child in node.children.items():
                _collect_words(child, current_word + char)

        _collect_words(self.root, "")
        return words

    def get_words_with_prefix(self, prefix: str) -> list:
        # 获取以指定前缀开头的所有单词
        words = []
        node = self.root
        for char in prefix:
            if char not in node.children:
                return []
            node = node.children[char]

        def _collect_words_with_prefix(node, current_prefix):
            for char, child in node.children.items():
                new_prefix = current_prefix + char
                if child.is_end_of_word:
                    words.append(new_prefix)
                _collect_words_with_prefix(child, new_prefix)

        _collect_words_with_prefix(node, prefix)
        return words
```

### 使用示例和测试代码

```python
if __name__ == "__main__":
    trie = Trie()

    # 插入单词
    trie.insert("apple")
    trie.insert("app")
    trie.insert("banana")
    trie.insert("applet")
    trie.insert("application")

    # 测试搜索
    print(trie.search("apple"))      # True
    print(trie.search("app"))        # True
    print(trie.search("appl"))       # False
    print(trie.search("banana"))     # True
    print(trie.search("orange"))     # False

    # 测试前缀存在性
    print(trie.starts_with("app"))     # True
    print(trie.starts_with("ban"))     # True
    print(trie.starts_with("appli"))   # True
    print(trie.starts_with("or"))      # False

    # 获取所有单词
    print(trie.get_all_words())  # ['app', 'apple', 'applet', 'application', 'banana']

    # 删除单词
    trie.delete("app")
    print(trie.search("app"))        # False
    print(trie.get_all_words())  # ['apple', 'applet', 'application', 'banana']

    # 获取具有指定前缀的单词
    print(trie.get_words_with_prefix("app"))  # ['apple', 'applet', 'application']

    print(trie.get_words_with_prefix("appli"))  # ['application']
```

### 复杂度分析

- **插入（Insert）**:
  - 时间复杂度：O(L)，其中 L 是单词的长度。
  - 空间复杂度：O(L)，每个字符可能需要创建一个新的节点。

- **搜索（Search）**:
  - 时间复杂度：O(L)，查找路径长度为 L。
  - 空间复杂度：O(1)，仅使用辅助变量。

- **前缀检查（Starts_with）**:
  - 时间复杂度：O(L)，查找前缀路径。
  - 空间复杂度：O(1)。

- **删除（Delete）**:
  - 时间复杂度：O(L)，需要遍历到单词的结尾，并检查是否可以删除节点。
  - 空间复杂度：O(1)，仅使用递归栈空间，与 L 成正比。

- **获取所有单词（get_all_words）**:
  - 时间复杂度：O(N)，其中 N 是Trie中存储的所有字符总数。
  - 空间复杂度：O(N)，存储所有单词。

- **获取具有指定前缀的单词（get_words_with_prefix）**:
  - 时间复杂度：O(M + N)，其中 M 是前缀长度，N 是以该前缀开头的字符总数。
  - 空间复杂度：O(N)，存储匹配的单词。

### 总结

Trie 树在处理字符串集合时具有显著优势，尤其适用于具有公共前缀的字符串存储和查找场景。其主要特点是在插入、搜索和前缀查询时，时间复杂度都与字符串长度呈线性关系，远优于哈希表在处理长字符串时的平均性能。在实现时，需要注意删除操作的特殊性，避免误删共享路径的节点。
